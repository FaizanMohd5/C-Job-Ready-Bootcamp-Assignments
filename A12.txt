1. Write a recursive function to print first N natural numbers
ans)
#include<stdio.h>
void printN(int n)
{
    if(n>0)
    {
        printN(n-1);
        printf("%d ",n);
    }
}
int main()
{
    int n;
    printf("Enter range:");
    scanf("%d",&n);
    printN(n);
    return 0;
}

10. Write a recursive function to print reverse of a given number
ans)
#include<stdio.h>
void rev(int n)
{
    if(n>=0 && n<=9)
    {
        printf("%d",n);
        return;
    }
    if(n>0)
    {
    printf("%d",n%10);
    rev(n/10);
    }
}
int main()
{
    int n;
    printf("Enter a number:");
    scanf("%d",&n);
    rev(n);
    return 0;
}
