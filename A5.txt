1. Write a program to print MySirG N times on the screen
ans)
#include<stdio.h>
int main()
{
    int i,n;
    printf("Enter a number: ");
    scanf("%d",&n);
    for(i=1;i<=n;i++)
            printf("MySirG\n");
    return 0;
}

ans)

{

    for(i=1;i<=n;i++)
            printf("%d ",i);
    return 0;
}

3. Write a program to print the first N natural numbers in reverse order
ans)
#include<stdio.h>
int main()
{
    int i,n;
    printf("Enter a number: ");
    scanf("%d",&n);
    for(i=n;i>=1;i--)
            printf("%d ",i);
    return 0;
}

10. Write a program to print a table of N.
ans)
#include<stdio.h>
int main()
{
    int i,n;
    printf("Enter a number: ");
    scanf("%d",&n);
    for(i=1;i<=10;i++)
    {
        printf("%d x %d = %d\n",n,i,n*i);
    }
    return 0;
}
